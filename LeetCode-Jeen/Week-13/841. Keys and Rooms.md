class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        int n = rooms.size();
        boolean[] visited = new boolean[n];

        dfs(rooms, 0, visited); // Start DFS from room 0

        for (boolean v : visited) {
            if (!v) return false;  // If there's any room we didn't visit, return false
        }

        return true;
    }

    private void dfs(List<List<Integer>> rooms, int curr, boolean[] visited) {
        if (visited[curr]) return; // If we've visited this room already, return
        visited[curr] = true;     // Mark current room as visited

        for (int key : rooms.get(curr)) {  // For each key in the room, visit the corresponding room
            dfs(rooms, key, visited);
        }
    }
}
