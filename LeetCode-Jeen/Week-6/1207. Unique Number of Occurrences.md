class Solution {
    public boolean uniqueOccurrences(int[] arr) {
        Arrays.sort(arr);
        
        int[] count = new int[arr.length];
        int uniqueCount = 0;
        
        count[0] = 1;
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] == arr[i - 1]) {
                count[uniqueCount]++;
            } else {
                uniqueCount++;
                count[uniqueCount] = 1;
            }
        }
        
        Arrays.sort(count, 0, uniqueCount + 1);
        for (int i = 1; i <= uniqueCount; i++) {
            if (count[i] == count[i - 1]) {
                return false;
            }
        }
        
        return true;     
    }
}
