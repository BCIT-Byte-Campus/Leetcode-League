class Solution {
    public boolean closeStrings(String word1, String word2) {
        if (word1.length() != word2.length()) {
            return false;
        }

        int[] count1 = new int[26];
        int[] count2 = new int[26];
        int[] exists1 = new int[26];
        int[] exists2 = new int[26];

        for (int i = 0; i < word1.length(); i++) {
            count1[word1.charAt(i) - 'a']++;
            count2[word2.charAt(i) - 'a']++;
            exists1[word1.charAt(i) - 'a'] = 1;
            exists2[word2.charAt(i) - 'a'] = 1;
        }

        Arrays.sort(count1);
        Arrays.sort(count2);

        return Arrays.equals(count1, count2) && Arrays.equals(exists1, exists2);   
    }
}
