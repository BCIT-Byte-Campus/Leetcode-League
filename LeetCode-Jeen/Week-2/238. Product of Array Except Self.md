class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] prefix_products = new int[n];
        int[] suffix_products = new int[n];
        int[] answer = new int[n];

        prefix_products[0] = 1;
        suffix_products[n-1] = 1;

        for (int i = 1; i < n; i++) {
            prefix_products[i] = nums[i-1] * prefix_products[i-1];
        }

        for (int i = n-2; i >= 0; i--) {
            suffix_products[i] = nums[i+1] * suffix_products[i+1];
        }

        for (int i = 0; i < n; i++) {
            answer[i] = prefix_products[i] * suffix_products[i];
        }

        return answer;
    }
}
