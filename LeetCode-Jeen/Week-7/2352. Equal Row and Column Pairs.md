class Solution {
    public int equalPairs(int[][] grid) {
        int count = 0;
        int n = grid.length;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int[] row = grid[i];
                int[] column = new int[n];

                for (int k = 0; k < n; k++) {
                    column[k] = grid[k][j];
                }

                if (Arrays.equals(row, column)) {
                    count++;
                }
            }
        }
        
        return count;
    }
}
