class Solution {
    public void moveZeroes(int[] nums) {
        int zeroPos = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != 0) {
                int temp = nums[zeroPos];
                nums[zeroPos] = nums[i];
                nums[i] = temp;
                zeroPos++;
            }
        }
    }
}
