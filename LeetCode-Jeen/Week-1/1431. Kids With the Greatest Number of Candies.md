class Solution {
    public List<Boolean> kidsWithCandies(int[] candies, int extraCandies) {
        List<Boolean> result = new ArrayList<>();
        for (int i = 0; i < candies.length; i++) {
            int max = candies[i];
            for (int j = 0; j < candies.length; j++) {
                if (candies[j] >= max) {
                    max = candies[j];
                }
            }
            if ((candies[i] + extraCandies) >= max) {
                result.add(true);
            } else {
                result.add(false);
            }
        }
        return result;
    }
}
