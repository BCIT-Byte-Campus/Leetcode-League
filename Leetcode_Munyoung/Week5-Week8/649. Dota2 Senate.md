class Solution {
    public String predictPartyVictory(String senate) {

        //if string senate is null or 0, it throws illegalArgumentException with the message 'Invalid input' 
        if(senate == null || senate.length() ==0) throw new IllegalArgumentException("Invalid input."); 
        final int N = senate.length(); 

        Queue <Integer> queR = new ArrayDeque<>(); //store the R index
        Queue <Integer> queD = new ArrayDeque<>(); //store the D index 

        //iterates each string senate 
        for(int i = 0; i < N; i++){

            //
            if(senate.charAt(i) == 'R'){
                queR.add(i); 
            } else {
                queD.add(i); 
            }
        }

        //While loop until two queue is empty. 
        while (!queR.isEmpty() && !queD.isEmpty()){
            int r = queR.poll(); 
            int d = queD.poll(); 

            //if r is smaller than d, r can ban the nearest d. 
            if(r<d){
                queR.add(r+N); //Wrapping arount to the end of the senate string 

                //if d is smaller than r, d can ban the nearest r. 
            }else{
                queD.add(d+N); 
            }
        }

        //if queR is empty, it means dire team is a winning team.
        //if not Radiant team is a winning team. 
        return queR.isEmpty() ? "Dire" : "Radiant"; 
    }
}
