class Solution {
    public int longestSubarray(int[] nums) {
        //Initialize  three variables
        int cur = 0;
        int prev = 0;
        int ans = 0;

        //Use for loop through nums array. 
        for(int i : nums){
            //If current element is equal to 1, increment cur. 
            if(i == 1) cur++;
            else{
                ans = Math.max(ans, cur + prev);
                prev = cur;
                cur = 0; 
            }
        }

        //After the loop, update ans by comparing the current ans and the sum of cur and prev. 
        ans = Math.max(ans, cur + prev); 

        //Check if ans is eqaul to the length of the nums array. 
        return ans == nums.length?ans-1:ans; 
    }
}
