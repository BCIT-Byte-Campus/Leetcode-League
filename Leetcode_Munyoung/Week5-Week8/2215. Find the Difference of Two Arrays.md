class Solution {
    public List<List<Integer>> findDifference(int[] nums1, int[] nums2) {

    //Create two hash set. 
      HashSet<Integer> set1 = new HashSet();
      HashSet<Integer> set2 = new HashSet();

    //Elements of nums1 are added to set1 
    //and elements of nums2 are added to set2. 
     for(int ele: nums1){
         set1.add(ele);
     }

     for(int ele:nums2){
         set2.add(ele);
     }

    //Create two array lists to store set1 and set2. 
      List<List<Integer>> list  new ArrayList<>();
      ArrayList<Integer> l1 = new ArrayList<>();
      ArrayList<Integer> l2 = new ArrayList<>();

    //Iterate element in set2. 
     for(int ele:set2){
        //If set1 doesn't contain el, then ele is added to l1. 
         if(set1.contains(ele) == false){
           l1.add(ele);
         }
     }

    //Iterate element in set1.
       for(int ele:set1){
        //If set2 doesn't cotain ele, the ele is added to l2. 
         if(set2.contains(ele)==false){
           l2.add(ele);
         }
     }


       list.add(l2);
       list.add(l1);
       return list;
    }
}
