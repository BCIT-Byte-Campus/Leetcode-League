class Solution {
    public int longestOnes(int[] nums, int k) {
        //Initilaize variable l to save current index of array nums. 
        int l = 0; 
        //Initialize variable zeros to save the number of 0. 
        int zeros = 0; 
        //Initialzie the variable res to save the max length of consecutive 1. 
        int res = Integer.MIN_VALUE; 

        //Use for loop through all elements of the nums array. 
        for(int r = 0; r <nums.length; r++){
            //If num array element is 0, increment the zeros. 
            if(nums[r] == 0){
                zeros++; 
            }

            //If the zeros is greater than k, it means the array has exceeded the allowed number of zeros. 
            if(zeros > k ){
                while(zeros > k){
                    if(nums[l] == 0) zeros--; 
                    l++; 
                }
            }

            //If the zeros is less than or equal to k, compare the two variables. 
            if(zeros <= k){
                res = Math.max(r-l+1, res); 
            }
        }
        return res; 
    }
}
