class Solution {
    public int findCircleNum(int[][] isConnected) {

        int n=isConnected.length;
        ArrayList<ArrayList<Integer>>adj=new ArrayList<>();
        
        for(int i=0;i<=n;i++){
            adj.add(new ArrayList<>());
        }

        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(isConnected[i][j]==1 && i!=j){
                    adj.get(i+1).add(j+1);
                }
            }
        }

        System.out.println(adj);

        int visit[]=new int[n+1];

        int count=0;
        for(int i=1;i<=n;i++){
            if(visit[i]==0){
                dfs(i,adj,visit);
                count++;
            }
        }
        return count;
    }

    public void dfs(int start,ArrayList<ArrayList<Integer>>adj,int[]visit){
        visit[start]=1;

        for(int node:adj.get(start)){
            if(visit[node]==0){
                dfs(node,adj,visit);
            }
        }
    }
}
