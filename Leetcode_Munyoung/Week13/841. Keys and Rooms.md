class Solution {
    public boolean canVisitAllRooms(List<List<Integer>> rooms) {
        
        int n = rooms.size();
        boolean[] visited = new boolean[n];
        Queue<Integer> q = new LinkedList<>();
        q.add(0); // we assume that we have 0th room's key
        
        while(!q.isEmpty())
        {
            int curr = q.poll();
            visited[curr] = true;
            for(int el : rooms.get(curr))
            {
                if(!visited[el])
                q.add(el);
            }
        }
        
        
        
        
        for(boolean el : visited)
        {
            if(!el) return false;
        }
        return true;
    }
}
