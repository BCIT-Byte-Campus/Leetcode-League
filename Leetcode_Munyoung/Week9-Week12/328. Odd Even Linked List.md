```java
class Solution {
    public ListNode oddEvenList(ListNode head) {
        // if given LinkedList has no nodes or single node then return head
        if(head == null || head.next == null){
            return head;
        }

        //Pointer to the Odd Elements
        ListNode curOdd = head; 
        //Pointer to the first even element 
        ListNode EvenList = head.next; 
        //Pointer to the Even Elements 
        ListNode curEven = EvenList; 

        while(curOdd.next != null && curEven.next != null){
            //currOdd will point to all Odd Elements iteratively
            curOdd.next = curEven.next;
            //moving the curOdd to next Odd Element
            curOdd = curOdd.next; 

            //currOdd will point to all Right Elements iteratively
            curEven.next = curOdd.next;
            //moving the curOdd to next Right Element
            curEven = curEven.next;
        }

        //Linking the odd Elements will Even Elements
        curOdd.next = EvenList;
        
        //returning the head
        return head; 
    }
}
```
