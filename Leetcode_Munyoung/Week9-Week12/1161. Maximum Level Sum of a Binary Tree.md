## My solution

```java
class Solution {
    public int maxLevelSum(TreeNode root) {
        // Initialize the queue to hold the nodes of the trees. 
        Queue<TreeNode> pq=new LinkedList();
        if(root==null)return 0;
        pq.add(root);
        int max=Integer.MIN_VALUE;

        //Initialize the level to 1: keep track of the current level being prcessed. 
        int level=1; 
        int maxlevel=1;

        //While the queue is not empty, 
        while(!pq.isEmpty()){
            //get the size of the queue
            int s=pq.size();
            //initialize a variable sum to 0: keep track of the sum of the nodes at the current level.
            int sum=0;
            for(int i=0;i<s;i++){
                TreeNode node=pq.remove();
                sum+=node.val;
                if(node.left!=null)pq.add(node.left);
                if(node.right!=null)pq.add(node.right);
            }

            //If sum is greater than max, update max to sum and maxlevel to level 
           if(sum>max){
               max=sum;
               maxlevel=level;
           }
          level++;
        }
        //return maxlevel
        return maxlevel;

    }
}
```
