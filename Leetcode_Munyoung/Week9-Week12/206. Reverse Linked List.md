```java
class Solution {
    public ListNode reverseList(ListNode head) {

        //Initialize temp and head2 as null 
        ListNode temp = null;
        ListNode head2 = null;

        //Traverse the linked list while head is not null
        while(head != null){
            //Save the reference to the current node in the temp variable. 
            temp = head;

            //Move the head pointer to the next node
            head = head.next;

            //Point the next pointer of the current node to the head2.
            temp.next = head2; 

            //Move the head2 pointer to the current node
            head2 = temp; 
        }

        //When the loop is finished head2 will be pointing to the new head of the reversed linked list
        return head2; 
    }
}
```
