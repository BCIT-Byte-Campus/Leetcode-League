## [Two Pointers] 1679. Max Number of K-Sum Pairs 

### My solution
Runtime: 620 ms

```py
class Solution(object):
    def maxOperations(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: int
        """
        nums.sort()  
        start, finish = 0, len(nums) - 1
        count = 0
        while start < finish:
            if nums[start] + nums[finish] == k:
                count += 1
                start += 1
                finish -= 1
            elif nums[start] + nums[finish] < k:
                start += 1
            else:
                finish -= 1
        return count
```

class Solution(object):
    def maxOperations(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: int
        """
        dic = defaultdict(int)
        ops = 0
        for x in nums:
            if x>=k:continue
            comp = k - x
            if dic[comp]>0:
                dic[comp]-=1
                ops+=1
            else:
                dic[x] +=1
        return ops
