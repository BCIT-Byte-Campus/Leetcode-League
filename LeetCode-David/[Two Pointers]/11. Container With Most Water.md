## [Two Pointers] 11. Container With Most Water

### My solution
Runtime: 617 ms

```py
class Solution(object):
    def maxArea(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        area = 0
        left = 0
        right = len(height) - 1

        while left < right:
            h = min(height[left], height[right])
            w = right - left
            area = max(area, h * w)

            if height[left] < height[right]:
                left += 1
            else:
                right -= 1

        return area
```

class Solution(object):
    def maxArea(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        ans, i, j = 0, 0, len(height)-1
        while (i < j):
            if height[i] <= height[j]:
                res = height[i] * (j - i)
                i += 1
            else:
                res = height[j] * (j - i)
                j -= 1
            if res > ans: ans = res
        return ans
