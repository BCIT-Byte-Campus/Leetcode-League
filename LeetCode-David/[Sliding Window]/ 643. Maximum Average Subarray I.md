## [Sliding Window] 643. Maximum Average Subarray I 

### My solution
Runtime: 1026 ms

```py
class Solution(object):
    def findMaxAverage(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: float
        """
        n = len(nums)
        sum_val = sum(nums[:k])
        max_sum = sum_val

        for i in range(k, n):
            sum_val = sum_val - nums[i-k] + nums[i]  
            max_sum = max(max_sum, sum_val)  

        return max_sum / float(k)
```

class Solution(object):
    def findMaxAverage(self, nums, k):
        if len(nums) < k:
            return -1
        slidingwindow = sum(nums[0:k])
        maxaverage = slidingwindow
        for i in range(k, len(nums)):
            slidingwindow = slidingwindow + nums[i] - nums[i-k]
            if slidingwindow > maxaverage:
                maxaverage = slidingwindow
        return float(maxaverage)/k
