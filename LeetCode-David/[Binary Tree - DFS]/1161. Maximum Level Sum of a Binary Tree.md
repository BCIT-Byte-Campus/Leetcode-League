## [Binary Tree - DFS] 1161. Maximum Level Sum of a Binary Tree
### My solution

```py
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxLevelSum(self, root: Optional[TreeNode]) -> int:
        ans, max_sum = -1, -float("inf")
        root, cur_level = [root], 1
        while root:
            nq, cur_sum = [], 0
            for cur_node in root:
                cur_sum += cur_node.val
                if cur_node.left:
                    nq.append(cur_node.left)
                if cur_node.right:
                    nq.append(cur_node.right)
            if cur_sum > max_sum:
                max_sum = cur_sum
                ans = cur_level
            cur_level += 1
            root = nq
        return ans
```
