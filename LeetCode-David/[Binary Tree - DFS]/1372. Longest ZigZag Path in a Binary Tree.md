## [Binary Tree - DFS] 1372. Longest ZigZag Path in a Binary Tree

### My solution

```py
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def longestZigZag(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if root == None:
            return 0
        self.max_ = 0
        self.dfs(root, 0, 0)
        return self.max_
    
    def dfs(self, root, prev, depth):
        self.max_ = max(depth, self.max_)
        if root.left:
            if prev == 0:
                self.dfs(root.left, 0, 1)
            else:
                self.dfs(root.left, 0, depth + 1)
        if root.right:
            if prev == 1:
                self.dfs(root.right, 1, 1)
            else:
                self.dfs(root.right, 1, depth + 1)
```
