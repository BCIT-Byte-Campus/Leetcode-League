## [Hash Map / Set] 1657. Determine if Two Strings Are Close

### My solution

```py
class Solution(object):
    def closeStrings(self, word1, word2):
        """
        :type word1: str
        :type word2: str
        :rtype: bool
        """
        if len(word1) != len(word2):
            return False

        def get_frequency_map(word):
            frequency_map = {}
            for char in word:
                if char in frequency_map:
                    frequency_map[char] += 1
                else:
                    frequency_map[char] = 1
            return frequency_map

        frequency_map1 = get_frequency_map(word1)
        frequency_map2 = get_frequency_map(word2)

        if set(frequency_map1.keys()) != set(frequency_map2.keys()):
            return False

        frequency_count1 = get_frequency_map(frequency_map1.values())
        frequency_count2 = get_frequency_map(frequency_map2.values())

        if frequency_count1 != frequency_count2:
            return False

        return True
```
